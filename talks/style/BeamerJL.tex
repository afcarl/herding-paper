    \large

  }
  \end{minipage}
}

\newcommand{\columnheader}[2]{
  \begin{minipage}[b]{#1}
  {
  \small
  \begin{center}
    \textbf{#2}
    \end{center}
    \vspace{-2.3\baselineskip}
    \begin{center}
      {\color{beamerblue} \rule{\linewidth}{1pt}}
    \end{center}
  }
  \end{minipage}
}

}
